# 매크로 상수
- `#define [매크로 상수 이름] [치환 값]`
```c
#define PI 3.141592

int main()
{
  int num = PI; // 3.141592 할당

  return 0;
}
```

# 매크로 함수
## 한 줄 매크로 함수
- `#define [매크로 함수 이름] [함수 기능]`
```c
#define SUM(x, y) x+y
#define MUL(x, y) x*y

int main()
{
  SUM(2, 3); // 5
  MUL(1+1, 2+1); // 6

  return 0;
}
```

# 전처리기 조건 명령어
## #if  #elif  #else  #endif
- 여러 조건 식에 따라 컴파일 되는 부분을 설정
```c
#include <stdio.h>

// 매크로 상수
# define OS 0

int main()
{
#if(OS == 0)
  print(0);

#elif(OS == 1)
  printf(1);

#else
  printf("X");

#endif
  return 0;
}
```

## #if defined
- 명령어가 정의되어 있는지 확인할 때 사용
```c
#include <stdio.h>

#define CAR

int main()
{
#if (defined CAR)
  printf("CAR O");

#else
  // == elif(!(defined CAR))
  printf("CAR X");

#endif
  return 0;
}
```

## #ifdef #ifndef
- `#if defined, #if !(defined)`와 동일
- 논리연산자 (&& ||) **사용 불가능**
```c
#define MAX 100

// 가능
#if ((defined MIN) && defined MAX)
// MIN과 MAX가 정의되어있으면 중간값 매크로 함수 만들기
#define MID ((MIN+MAX)/2)
#endif

// 불가능
#ifdef MIN && MAX
#define MID ((MIN+MAX)/2)
#endif
```

# 헤더파일.h
- 선언만 하는 파일, 소스 파일에 포함시킴
- 헤더파일을 새로 포함할 때: `#include "header.h"`

## 중복 컴파일 문제 예방
```c
// main.c

#include <stdio.h>
#include "myheader.h" // 헤더파일을 포함할 것

int main()
{
  doulbe r = 3.3;
  printf(circle(r)); //원의 지름
  printf(area(r)); // 원의 면적
}
```

- 헤더파일에서 중복 컴파일 예방
```c
// myheader.h

// INCLUDED_MYHEADER라는 매크로가 정의되어있지 않으면
#ifndef INCLUDED_MYHEADER
// 해당 매크로를 정의해라
#define INCLUDED_MYHEADER

// 해당 매크로 ~~
#define PI 3.141592

// 함수명만 선언, 정의는 안함
double circle(double radius);
double area(double radius);

// ~~ 여기까지
#endif
```

```c
// func.c

// 헤더파일을 포함할 것
#include "myheader.h"

// 헤더파일과 달리 함수 기능 정의
double circle(double radius)
{
    double result = 2.0 * PI * radius;
    return result;
}

double area(double radius)
{
    double result = PI * radius * radius;
    return result;
}
```
